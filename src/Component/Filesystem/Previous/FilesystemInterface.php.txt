<?php

declare(strict_types=1);

namespace Laventure\Component\Filesystem\Previous;

/**
 * FilesystemInterface
 *
 * @author Jean-Claude <jeanyao@ymail.com>
 *
 * @license https://github.com/jeandev84/laventure-framework/blob/master/LICENSE
 *
 * @package  Laventure\Component\Filesystem
 */
interface FilesystemInterface
{


     /**
      * @param string $file
      * @return mixed
     */
     public function locate(string $file): mixed;





     /**
      * @param string $file
      * @param string $content
      * @return mixed
     */
     public function write(string $file, string $content): mixed;





     /**
      * @param string $file
      * @return mixed
     */
     public function read(string $file): mixed;






     /**
      * @param string $resource
      * @return mixed
     */
     public function stream(string $resource): mixed;
}
